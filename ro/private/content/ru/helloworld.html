Этот небольшой пример демонстрирует способность Pusa работать исключительно с кодом backend.

/* без спицифического кода */
/* Pusa SDK */ $this -> action /* определили новое действие */ ( 'helloworld-click', /* id действия */ [ $this -> copy() /* создали ветку директив */ -> alert( 'Hello World' ) /* указали директиву alert */ ] ) -> body() /* установили фокус на body */ -> children( [ 'equal', '#id', 'helloworld' ]) /* сместили фокус на элемент по id */ -> event( 'click', 'helloworld-click' ); /* назначили действие на событие */
<button id="helloworld"> Click me </button>

Pusa использует декларативный формат директив, который формируется во время выполнения методов на сервере.

[ [ "action", "helloworld-click", [ [ "alert", "Hello world" ] ] ], [ "body" ], [ "children", [ "equal", "#id", "helloworld" ] ], [ "event", "click", "helloworld-click" ] ]
